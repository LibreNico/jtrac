43/**
44 * XML metadata is one of the interesting design decisions of JTrac.
45 * Metadata is defined for each space and so Items that belong to a
46 * space are customized by the space metadata.  This class can marshall
47 * and unmarshall itself to XML and this XML is stored in the database
48 * in a single column.  Because of this approach, Metadata can be made more
49 * and more complicated in the future without impact to the database schema.
50 *
51 * Things that the Metadata configures for a Space:
52 * 
53 * 1) custom Fields for an Item (within a Space)
54 * - Label
55 * - whether mandatory or not [ DEPRECATED ]
56 * - the option values (drop down list options)
57 * - the option "key" values are stored in the database (WITHOUT any relationships)
58 * - the values corresponding to "key"s are resolved in memory from the Metadata
59 *   and not through a database join.
60 *
61 * 2) the Roles available within a space
62 * - for each (from) State the (to) State transitions allowed for this role
63 * - and within each (from) State the fields that this Role can view / edit
64 *
65 * 3) the State labels corresponding to each state 
66 * - internally States are integers, but for display we need a label
67 * - labels can be customized
68 * - special State values: 0 = New, 1 = Open, 99 = Closed
69 *
70 * 4) the order in which the fields are displayed
71 * on the data entry screens and the query result screens etc.
72 *
73 * There is one downside to this approach and that is there is a limit
74 * to the nunmbers of custom fields available.  The existing limits are
75 * - Drop Down: 10

404        State state = getRoleState(roleKey, stateKey);
405        Field.Name tempName = Field.convertToName(fieldName);        
406        Integer mask = state.getFields().get(tempName);
407        switch(mask) {
408            // case State.MASK_HIDDEN: state.getFields().put(name, State.MASK_READONLY); return; HIDDEN support in future
409            case State.MASK_READONLY: state.getFields().put(tempName, State.MASK_OPTIONAL); return;
410            case State.MASK_OPTIONAL: state.getFields().put(tempName, State.MASK_MANDATORY); return;            
411            case State.MASK_MANDATORY: state.getFields().put(tempName, State.MASK_READONLY); return;
412            default: // should never happen
413        }
414    }

420    public List<Field> getEditableFields(Collection<String> roleKeys, int status) {
421        Map<Field.Name, Field> fs = new HashMap<Field.Name, Field>(getFieldCount());     
422        for(String roleKey : roleKeys) {
423            if (roleKey.startsWith("ROLE_")) {
424                continue;
425            }
426            if(status > -1) {
427                State state = getRoleState(roleKey, status);
428                fs.putAll(getEditableFields(state));  
429            } else { // we are trying to find all editable fields
430                Role role = roles.get(roleKey);                
431                for(State state : role.getStates().values()) {
432                    if(state.getStatus() == State.NEW) {
433                        continue;
434                    }
435                    fs.putAll(getEditableFields(state));
436                }                
437            }
438        }
439        // just to fix the order of the fields
440        List<Field> result = new ArrayList<Field>(getFieldCount());
441        for(Field.Name fieldName : fieldOrder) {
442            Field f = fs.get(fieldName);
443            // and not all fields may be editable
444            if(f != null) {
445                result.add(f);
446            }
447        }
448        return result;
449    }

455    private Map<Field.Name, Field> getEditableFields(State state) {
456        Map<Field.Name, Field> fs = new HashMap<Field.Name, Field>(getFieldCount());
457        for(Map.Entry<Field.Name, Integer> entry : state.getFields().entrySet()) {
458            if (entry.getValue() == State.MASK_OPTIONAL || entry.getValue() == State.MASK_MANDATORY) {
459                Field f = fields.get(entry.getKey());
460                // set if optional or not, this changes depending on the user / role and status
461                f.setOptional(entry.getValue() == State.MASK_OPTIONAL);
462                fs.put(f.getName(), f);
463            }
464        }
465        return fs;
466    }
