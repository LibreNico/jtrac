504        HSSFWorkbook wb = null;
505        try {
506            fs = new POIFSFileSystem(is);

508        } catch (Exception e) {
509            throw new RuntimeException(e);
510        }        
511        HSSFSheet sheet = wb.getSheetAt(0);
512        HSSFRow r = null;

514        int row = 0;
515        int col = 0;
516        columns = new ArrayList<Column>();
517        //========================== HEADER ====================================
518        // column headings are important, this routine assumes that the first
519        // row is a header row and that reaching an empty cell means end of data
520        r = sheet.getRow(row);       
521        while(true) {            
522            c = r.getCell((short) col);
523            if (c == null) {          
524                break;
525            }
526            String value = c.getStringCellValue();
527            if (value == null || value.trim().length() == 0) {
528                break;
529            }
530            Column column = new Column(value.trim());
531            columns.add(column);
532            col++;
533        }
534        //============================ DATA ====================================
535        rows = new ArrayList<List<Cell>>();
536        while(true) {
537            row++;

541            }

551                        value = c.getNumericCellValue();
