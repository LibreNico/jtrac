578                    continue;
579                }
580                UserSpaceRole temp = new UserSpaceRole(user, space, roleKey);
581                if(!allocated.contains(temp)) {
582                    notFullyAllocated.add(user);
583                    break;
584                }
585            }
586        }

750    public void rebuildIndexes(BatchInfo batchInfo) {
751        File file = new File(jtracHome + "/indexes");
752        for (File f : file.listFiles()) {
753            logger.debug("deleting file: " + f);
754            f.delete();
755        }        
756        logger.info("existing index files deleted successfully");
757        int totalSize = dao.loadCountOfAllItems();
758        batchInfo.setTotalSize(totalSize);
759        logger.info("total items to index: " + totalSize);                
760        int firstResult = 0;
761        long lastFetchedId = 0;
762        while(true) {
763            logger.info("processing batch starting from: " + firstResult + ", current: " + batchInfo.getCurrentPosition());
764            List<Item> items = dao.findAllItems(firstResult, batchInfo.getBatchSize());
765            for (Item item : items) {                                    
766                
767                    indexer.index(item);  
768                
769                // currently history is indexed separately from item
770                // not sure if this is a good thing, maybe it gives
771                // more flexibility e.g. fine-grained search results
772                    
773                int historyCount = 0;
774                for(History history : item.getHistory()) {
775                    indexer.index(history);
776                    historyCount++;
777                }
778                if(logger.isDebugEnabled()) {
779                    logger.debug("indexed item: " + item.getId() 
780                            + " : " + item.getRefId() + ", history: " + historyCount);
781                }
782                batchInfo.incrementPosition();
783                lastFetchedId = item.getId();
784            }                        
785            if(logger.isDebugEnabled()) {
786              logger.debug("size of current batch: " + items.size());
787              logger.debug("last fetched Id: " + lastFetchedId);
788            }  
789            firstResult += batchInfo.getBatchSize();
790            if(logger.isDebugEnabled()) {
791                logger.debug("setting firstResult to: " + firstResult);
792            }
793            if(batchInfo.isComplete()) {
794                logger.info("batch completed at position: " + batchInfo.getCurrentPosition());
795                break;
796            }
797        }        
798        
799    }
